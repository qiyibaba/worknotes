create unique index tt_idx_2 on tt  (c) local;

CREATE INDEX tt_idx_5 ON tt(f) GLOBAL PARTITION BY hash(f) partitions 10 ;
CREATE unique INDEX tt_idx_6 ON tt(g) GLOBAL PARTITION BY hash(g) partitions 10 ;
